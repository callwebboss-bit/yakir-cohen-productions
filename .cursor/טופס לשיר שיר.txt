        state.pkg       = id;

        state.pkgName   = el.querySelector('h3').innerText;

        // ── FIX: Reset all options (state + UI) on package switch ──
        state.options = [];
        document.querySelectorAll('.option-card').forEach(card => {
            card.classList.remove('selected');
        });

        // ── FIX: Reset guests to 1 on package switch ──
        state.guests   = 1;
        state.editMode = 'choir';
        document.getElementById('guest-count').innerText    = '1';
        document.getElementById('visual-singers').innerText = '👤';
        document.getElementById('guest-warning').style.display  = 'none';
        document.getElementById('edit-switcher').style.display  = 'none';
        document.querySelectorAll('.es-opt').forEach(e => e.classList.remove('active'));
        document.querySelectorAll('.es-opt')[0].classList.add('active');

        // Update card highlight
        document.querySelectorAll('.pkg-card').forEach(c => c.classList.remove('active'));
        el.classList.add('active');

        // Strategy upsell: hide if included in Gold/VIP
        const stratEl = document.getElementById('opt-strat');
        if (id === 'gold' || id === 'vip') {
            stratEl.style.display = 'none';
            stratEl.classList.remove('selected');
        } else {
            stratEl.style.display = 'flex';
        }

        // Show configurator
        document.getElementById('configurator').style.display = 'block';
        document.getElementById('sticky-footer').classList.add('visible');
        document.getElementById('pBar').style.width = '33%';

        // Hide scroll hint after first interaction
        document.getElementById('scroll-hint').classList.add('hidden');

        setTimeout(() => window.scrollTo({ top: 200, behavior: 'smooth' }), 100);
        recalc();
    }

    // ─── TOGGLE OPTION ───────────────────────────────────────────
    function toggleOption(el, price, name) {
        el.classList.toggle('selected');
        const isSelected = el.classList.contains('selected');

        if (isSelected) {
            state.options.push({ name, price });
            // FIX: throttle confetti — max once per 500ms
            const now = Date.now();
            if (now - confettiLast > 500) {
                confetti({ particleCount: 30, spread: 50, origin: { y: 0.8 }, colors: ['#c5a059'] });
                confettiLast = now;
            }
        } else {
            state.options = state.options.filter(o => o.name !== name);
        }

        // FIX: haptic feedback on mobile
        if (navigator.vibrate) navigator.vibrate(10);

        document.getElementById('pBar').style.width = '66%';
        recalc();
    }

    // ─── GUESTS ──────────────────────────────────────────────────
    function updateGuests(delta) {
        const newVal = state.guests + delta;
        if (newVal < 1) return;
        state.guests = newVal;

        // FIX: haptic feedback
        if (navigator.vibrate) navigator.vibrate(8);

        // Update counter display
        document.getElementById('guest-count').innerText = state.guests;

        // Update emoji display
        let visuals = '';
        for (let i = 0; i < Math.min(state.guests, 5); i++) visuals += '👤';
        if (state.guests > 5) visuals += '➕';
        document.getElementById('visual-singers').innerText = visuals;

        // Show/hide warning and edit mode switcher
        const warning  = document.getElementById('guest-warning');
        const switcher = document.getElementById('edit-switcher');
        if (state.guests > 2) {
            warning.style.display  = 'block';
            switcher.style.display = 'block';
        } else {
            warning.style.display  = 'none';
            switcher.style.display = 'none';
            state.editMode = 'choir';
            document.querySelectorAll('.es-opt').forEach(e => e.classList.remove('active'));
            document.querySelectorAll('.es-opt')[0].classList.add('active');
        }

        // FIX: debounce recalc on rapid taps (150ms)
        clearTimeout(guestDebounce);
        guestDebounce = setTimeout(recalc, 150);
    }

    // ─── EDIT MODE ───────────────────────────────────────────────
    function setEditMode(mode, el) {
        state.editMode = mode;
        document.querySelectorAll('.es-opt').forEach(e => e.classList.remove('active'));
        el.classList.add('active');
        recalc();
    }

    // ─── RECALCULATE ─────────────────────────────────────────────
    function recalc() {
        let total = state.basePrice;

        // Add selected options
        state.options.forEach(o => total += o.price);

        // Guest surcharges
        let guestCost = 0;
        if (state.guests > 2) {
            const extra = state.guests - 2;
            guestCost += extra * 250;                              // base extra fee
            if (state.editMode === 'full') guestCost += extra * 290; // full-edit fee per extra person
        }
        total += guestCost;

        // FIX: animate from stored state value, not DOM string
        animateValue(document.getElementById('total-price'), state.currentTotal, total, 400);
        state.currentTotal = total;

        // Build details popup HTML
        let html = `<div class="details-row"><span>חבילה: ${state.pkgName}</span><span>₪${state.basePrice.toLocaleString()}</span></div>`;
        state.options.forEach(o => {
            html += `<div class="details-row"><span>${o.name}</span><span>+₪${o.price.toLocaleString()}</span></div>`;
        });
        if (guestCost > 0) {
            const modeLabel = state.editMode === 'full' ? ' (עריכה אישית)' : ' (מקהלה)';
            html += `<div class="details-row"><span>תוספת משתתפים${modeLabel}</span><span>+₪${guestCost.toLocaleString()}</span></div>`;
        }
        html += `<div class="details-row final"><span>סה"כ</span><span>₪${total.toLocaleString()}</span></div>`;
        document.getElementById('details-content').innerHTML = html;

        // Build WhatsApp message
        const optList = state.options.map(o => o.name).join(', ') || 'ללא';
        const guestInfo = state.guests > 2
            ? `${state.guests} (${state.editMode === 'full' ? 'עריכה מלאה' : 'מקהלה'})`
            : state.guests;
        const txt = `היי יקיר! בניתי חבילה באתר:%0A🎤 *חבילה:* ${state.pkgName}%0A➕ *תוספות:* ${optList}%0A👥 *משתתפים:* ${guestInfo}%0A💰 *סה"כ:* ₪${total.toLocaleString()}`;
        document.getElementById('wa-link').href = `https://wa.me/972587555456?text=${txt}`;

        // Mark configurator complete
        if (state.options.length > 0 || state.guests > 1) {
            document.getElementById('pBar').style.width = '90%';
        }
    }

    // ─── TOGGLE DETAILS POPUP ────────────────────────────────────
    function toggleDetails() {
        document.getElementById('details-popup').classList.toggle('show');
    }
    // Close popup when clicking outside
    document.addEventListener('click', (e) => {
        const popup = document.getElementById('details-popup');
        const total = document.querySelector('.total-container');
        if (!popup.contains(e.target) && !total.contains(e.target)) {
            popup.classList.remove('show');
        }
    });

    // ─── ANIMATE PRICE (FIX: cancel previous frame) ──────────────
    function animateValue(el, start, end, duration) {
        if (animFrame) cancelAnimationFrame(animFrame);
        if (start === end) {
            el.innerHTML = '₪' + end.toLocaleString();
            return;
        }
        let startTimestamp = null;
        const step = (timestamp) => {
            if (!startTimestamp) startTimestamp = timestamp;
            const progress = Math.min((timestamp - startTimestamp) / duration, 1);
            const eased    = 1 - Math.pow(1 - progress, 3); // ease-out cubic
            el.innerHTML   = '₪' + Math.floor(eased * (end - start) + start).toLocaleString();
            if (progress < 1) {
                animFrame = requestAnimationFrame(step);
            } else {
                el.innerHTML = '₪' + end.toLocaleString();
                animFrame = null;
            }
        };
        animFrame = requestAnimationFrame(step);
    }
