function selectPath(path) {
  document.querySelectorAll('.path-card').forEach(c => c.classList.remove('sel'));
  document.getElementById('path-' + path).classList.add('sel');

  const d = PATHS[path];
  buildCarousel(path);

  document.getElementById('dEyebrow').textContent = d.eyebrow;
  document.getElementById('dHeadline').innerHTML = d.headline;
  document.getElementById('dCopy').innerHTML = d.copy;
  document.getElementById('dFeatures').innerHTML = d.features.map(f => `<div class="feat">${f}</div>`).join('');
  document.getElementById('pkgSub').textContent = d.pkgSub;

  document.getElementById('pkgGrid').innerHTML = d.packages.map(p => {
    const msg = `היי יקיר! 👋 ${d.eyebrow} - מעניינת אותי חבילת ${p.name} (${p.price})`;
    const link = `https://wa.me/${WA}?text=${encodeURIComponent(msg)}`;
    return `
    <div class="pkg${p.featured ? ' feat-pkg' : ''}">
      ${p.badge ? `<div class="pkg-badge">${p.badge}</div>` : ''}
      <div class="pkg-name">${p.name}</div>
      <div class="pkg-price">${p.price}</div>
      <div class="pkg-period">${p.period}</div>
      <ul class="pkg-list">${p.includes.map(i => `<li>${i}</li>`).join('')}</ul>
      <a class="pkg-cta" href="${link}" target="_blank">שריין את התאריך שלי ←</a>
    </div>`;
  }).join('');

  const mainMsg = `https://wa.me/${WA}?text=${encodeURIComponent(d.waMsg)}`;
  document.getElementById('mainCTA').href = mainMsg;

  const det = document.getElementById('detail');
  det.classList.remove('show');
  void det.offsetWidth;
  det.classList.add('show');
  setTimeout(() => det.scrollIntoView({ behavior: 'smooth', block: 'start' }), 80);
}

  if (p.get('c') || p.get('type') || p.get('pay')) location.replace('portal.html' + location.search);
  if (p.get('d') || p.get('v')) location.replace('view.html' + location.search);
})();
