    navigator.share({
      title: `🎁 שובר מתנה עבור ${S.toName||''}`,
      text: msg,
      url: url
    }).catch(() => {});
  } else {
    /* Fallback: copy + open WA */

      toast('✅ הודעה הועתקה! פותח וואטסאפ...');

      window.open(`https://wa.me/?text=${encodeURIComponent(msg)}`, '_blank');
    });
  }
}

}

async function dlImg(fmt) {
  document.getElementById('imgMenu').classList.remove('open');
  toast('⏳ מייצר תמונה...');
  await document.fonts.ready;

  const sizes = { standard:{w:1200,h:800}, story:{w:1080,h:1920}, square:{w:1080,h:1080} };
  const { w, h } = sizes[fmt] || sizes.standard;
  const cv = document.createElement('canvas');
  cv.width = w; cv.height = h;
  const ctx = cv.getContext('2d');
  ctx.direction = 'rtl';

  const c = TPL_C[S.tpl] || TPL_C[0];
  const p = CFG.PKGS[S.pkg];
  const pd = w * 0.055; // padding

  /* Background */
  const bgG = ctx.createLinearGradient(0,0,w,h);
  bgG.addColorStop(0, c.bg1); bgG.addColorStop(1, c.bg2);
  ctx.fillStyle = bgG; ctx.fillRect(0,0,w,h);

  /* Radial glow */
  const gl = ctx.createRadialGradient(w*.18,h*.18,0,w*.18,h*.18,w*.6);
  gl.addColorStop(0,'rgba(201,168,76,0.07)'); gl.addColorStop(1,'transparent');
  ctx.fillStyle = gl; ctx.fillRect(0,0,w,h);

  /* Outer border */
  ctx.strokeStyle = c.border; ctx.lineWidth = 2;
  ctx.strokeRect(pd, pd, w-pd*2, h-pd*2);

  /* Corner brackets */
  const cs = w * 0.038;
  ctx.strokeStyle = c.gold; ctx.lineWidth = w * 0.003;
  [
    [pd, pd, 1, 1], [w-pd, pd, -1, 1],
    [pd, h-pd, 1, -1], [w-pd, h-pd, -1, -1]
  ].forEach(([x,y,dx,dy]) => {
    ctx.beginPath();
    ctx.moveTo(x, y + dy*cs); ctx.lineTo(x,y); ctx.lineTo(x + dx*cs, y);
    ctx.stroke();
  });

  const tx = w - pd - w*0.045; // right-side x for RTL text
  let y = pd + h * 0.1;

  /* Mic */
  ctx.font = `${w*0.055}px sans-serif`;
  ctx.fillText('🎙', pd + w*0.04, y);

  /* Brand (right-aligned) */
  ctx.textAlign = 'right'; ctx.textBaseline = 'middle';
  ctx.fillStyle = c.gold;
  ctx.font = `700 ${w*0.024}px 'Rajdhani','Heebo',sans-serif`;
  ctx.fillText('יקיר כהן הפקות', tx, y - w*0.01);
  ctx.fillStyle = c.muted;
  ctx.font = `${w*0.017}px 'Rajdhani','Heebo',sans-serif`;
  ctx.fillText('PRODUCTIONS', tx, y + w*0.018);

  /* Package badge */
  const bw = w * 0.22; const bh = h * 0.055;
  const bx = pd + w*0.04; const by = y - bh/2;
  ctx.fillStyle = `rgba(201,168,76,0.1)`;
  rr(ctx, bx, by, bw, bh, bh/2); ctx.fill();
  ctx.strokeStyle = c.border; ctx.lineWidth = 1;
  rr(ctx, bx, by, bw, bh, bh/2); ctx.stroke();
  ctx.fillStyle = c.gold;
  ctx.font = `700 ${w*0.019}px 'Rajdhani','Heebo',sans-serif`;
  ctx.textAlign = 'left'; ctx.textBaseline = 'middle';
  ctx.fillText(p.name, bx + bh*0.4, by + bh/2);

  y = h * (fmt === 'story' ? 0.28 : 0.33);

  /* "שובר מתנה עבור" */
  ctx.fillStyle = c.muted;
  ctx.font = `${w*0.022}px 'Heebo',sans-serif`;
  ctx.textAlign = 'right'; ctx.textBaseline = 'alphabetic';
  ctx.fillText('שובר מתנה עבור', tx, y);

  /* Recipient name */
  const nSize = Math.min(w * 0.092, h * (fmt==='story'?0.07:0.12));
  ctx.fillStyle = c.gold;
  ctx.font = `900 ${nSize}px 'Frank Ruhl Libre','Heebo',serif`;
  y += nSize * 1.1;
  ctx.fillText(S.toName || 'שמך כאן', tx, y);

  /* Package */
  y += w * 0.055;
  ctx.fillStyle = c.text;
  ctx.font = `700 ${w*0.028}px 'Heebo',sans-serif`;
  ctx.fillText(`${p.name} · ${p.price}`, tx, y);

  /* Desc */
  y += w * 0.038;
  ctx.fillStyle = c.muted;
  ctx.font = `${w*0.02}px 'Heebo',sans-serif`;
  ctx.fillText(p.desc, tx, y);

  /* Message */
  if (S.message) {
    y += w * 0.04;
    ctx.strokeStyle = c.border; ctx.lineWidth = w*0.004;
    ctx.beginPath();
    ctx.moveTo(tx - w*0.003, y - w*0.01);
    ctx.lineTo(tx - w*0.003, y + h*0.16);
    ctx.stroke();
    ctx.fillStyle = c.muted;
    ctx.font = `italic ${w*0.021}px 'Frank Ruhl Libre','Heebo',serif`;
    const lines = wrapT(ctx, S.message, w * 0.7);
    const lineH = w * 0.032;
    lines.slice(0,5).forEach((line,i) => {
      ctx.fillText(line, tx - w*0.025, y + i*lineH);
    });
    y += Math.min(lines.length,5)*lineH + w*0.025;
  }

  /* Footer line */
  const fy = h - pd - h*(fmt==='story'?0.08:0.1);
  ctx.strokeStyle = 'rgba(255,255,255,0.07)'; ctx.lineWidth = 1;
  ctx.beginPath();
  ctx.moveTo(pd+w*0.04, fy); ctx.lineTo(w-pd-w*0.04, fy); ctx.stroke();

  ctx.fillStyle = c.muted;
  ctx.font = `${w*0.019}px 'Heebo',sans-serif`;
  ctx.textAlign = 'right'; ctx.textBaseline = 'alphabetic';
  let fy2 = fy + h*0.04;
  if (S.datetime) { ctx.fillText(`📅 ${S.datetime}`, tx, fy2); fy2 += h*0.03; }
  if (S.address)  { ctx.fillText(`📍 ${S.address}`,  tx, fy2); }

  if (S.fromName) {
    ctx.textAlign = 'left'; ctx.textBaseline = 'alphabetic';
    ctx.fillStyle = c.muted;
    ctx.font = `${w*0.015}px 'Rajdhani','Heebo',sans-serif`;
    ctx.fillText('מאת', pd+w*0.04, fy + h*0.03);
    ctx.fillStyle = c.gold;
    ctx.font = `${w*0.022}px 'Frank Ruhl Libre','Heebo',serif`;
    ctx.fillText(S.fromName, pd+w*0.04, fy + h*0.06);
  }

  /* Download */
  const a = document.createElement('a');
  a.href = cv.toDataURL('image/png');
  a.download = `yakir-voucher-${(S.toName||'guest').replace(/\s/g,'-')}-${fmt}.png`;
  a.click();
  toast('✅ תמונה נשמרה!');
}

function rr(ctx,x,y,w,h,r){
  ctx.beginPath();ctx.moveTo(x+r,y);ctx.lineTo(x+w-r,y);
  ctx.quadraticCurveTo(x+w,y,x+w,y+r);ctx.lineTo(x+w,y+h-r);
  ctx.quadraticCurveTo(x+w,y+h,x+w-r,y+h);ctx.lineTo(x+r,y+h);
  ctx.quadraticCurveTo(x,y+h,x,y+h-r);ctx.lineTo(x,y+r);
  ctx.quadraticCurveTo(x,y,x+r,y);ctx.closePath();
}

function wrapT(ctx, text, maxW) {
  const words=text.split(' '); const lines=[]; let line='';
  words.forEach(w=>{
    const t=line?`${line} ${w}`:w;
    if(ctx.measureText(t).width>maxW && line){ lines.push(line);line=w; }
    else line=t;
  });
  if(line)lines.push(line);
  return lines;
}
