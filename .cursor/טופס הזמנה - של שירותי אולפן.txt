    /* ─── Package Context (above upsells) ─── */

    function _updatePkgContext() {

      const bar   = document.getElementById('pkg-context-bar');

      const nudge = document.getElementById('pkg-nudge');

      const cont  = document.getElementById('pkg-context-content');

      if (!bar || !nudge || !cont) return;



      let cartItems = [];

      try { cartItems = JSON.parse(localStorage.getItem('yk_cart_v3') || '[]'); } catch (_) {}

      const pkgs = cartItems.filter(i => i.isPackage !== false);



      if (pkgs.length > 0) {

        const p = pkgs[0];

        const cur = (window.YK_CONFIG && YK_CONFIG.studio && YK_CONFIG.studio.currency) || '₪';

        cont.innerHTML =

          '<div style="display:flex;justify-content:space-between;align-items:center;gap:8px;">' +

          '<span style="font-weight:700;font-size:.92rem;color:#000;">' + p.name + '</span>' +

          '<div style="display:flex;align-items:center;gap:12px;">' +

          '<span style="font-weight:800;color:#8B0000;font-size:.92rem;">' + cur + (p.price || 0).toLocaleString() + '</span>' +

          '<button onclick="YK_Cart.openDrawer()"' +

          ' style="font-size:.74rem;color:#7A7A7A;background:none;border:none;cursor:pointer;' +

          'font-family:\'Heebo\',sans-serif;text-decoration:underline;padding:0;">שנה</button>' +

          '</div></div>';

        bar.style.display   = 'block';

        nudge.style.display = 'none';

      } else {

        bar.style.display   = 'none';

        nudge.style.display = 'flex';

      }

    }



    document.addEventListener('yk:cartChanged', _updatePkgContext);

    document.addEventListener('yk:itemAdded',   _updatePkgContext);
