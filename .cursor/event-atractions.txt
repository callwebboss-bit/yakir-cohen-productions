const APP = (() => {

  const LS_KEY = 'peak_moments_v3'; // bump version to invalidate stale state

  // ── STATE ─────────────────────────────────────────────────────
  const state = {
    currentStep: 1,
    eventType:   null,
    eventLabel:  '',
    eventDate:   '',
    guests:      0,
    locationFee: 0,
    contact:     { name:'', phone:'', email:'', notes:'' },
    attractions: {
      smoke:false, co2:false, bubbles:false, sparks:false, confetti:false,
      backstage:false, podcast:false, led:false, sound:false, studio:false
    },
    subs: {
      smoke:'large', co2:'color', bubbles:'soap', sparks:'4u',
      confetti:'mega', 'confetti-paper':'white', backstage:'fun'
    },
    extras:     { hazer:false, laser:false, early:false, clip:false },
    extraPrices:{ hazer:600, laser:1100, early:800, clip:950 },
    crossSell:  { photo:false, dj:false }
  };

  // ── LOCAL STORAGE ─────────────────────────────────────────────
  function saveLS() {
    try {
      localStorage.setItem(LS_KEY, JSON.stringify({
        eventType: state.eventType, eventLabel: state.eventLabel,
        eventDate: state.eventDate, guests: state.guests,
        locationFee: state.locationFee, contact: state.contact,
        attractions: state.attractions, subs: state.subs,
        extras: state.extras, crossSell: state.crossSell,
        currentStep: state.currentStep
      }));
    } catch(e) {}
  }

  function loadLS() {
    try {
      const raw = localStorage.getItem(LS_KEY);
      if (!raw) return;
      const d = JSON.parse(raw);
      if (!d || typeof d !== 'object') return; // corrupt data guard
      if (d.eventType)  { state.eventType = d.eventType; state.eventLabel = d.eventLabel || ''; }
      if (d.eventDate)  state.eventDate = d.eventDate;
      if (d.guests)     state.guests = d.guests;
      if (d.locationFee)state.locationFee = d.locationFee;
      if (d.contact)    Object.assign(state.contact, d.contact);
      if (d.attractions)Object.assign(state.attractions, d.attractions);
      if (d.subs)       Object.assign(state.subs, d.subs);
      if (d.extras)     Object.assign(state.extras, d.extras);
      if (d.crossSell)  Object.assign(state.crossSell, d.crossSell);
      // Restore UI
      restoreUI();
      if (d.currentStep > 1) {
        state.currentStep = d.currentStep;
        showStep(state.currentStep);
        toast('ברוכים השבים! ממשיכים מהמקום שעצרתם 💾', 'info', 4000);
      }
    } catch(e) {}
  }

  function restoreUI() {
    // Event type
    if (state.eventType) {
      const el = document.getElementById('ev-' + state.eventType);
      if (el) el.classList.add('sel');
    }
    // Date, guests, location
    const evDate = document.getElementById('ev-date');
    if (evDate && state.eventDate) evDate.value = state.eventDate;
    const evGuests = document.getElementById('ev-guests');
    if (evGuests && state.guests) evGuests.value = state.guests;
    const evLoc = document.getElementById('ev-loc');
    if (evLoc && state.locationFee) evLoc.value = state.locationFee;
    // Contact
    const cn1 = document.getElementById('c-name-1');
    if (cn1 && state.contact.name) cn1.value = state.contact.name;
    const cp1 = document.getElementById('c-phone-1');
    if (cp1 && state.contact.phone) cp1.value = state.contact.phone;
    // Attractions
    Object.entries(state.attractions).forEach(([k, v]) => {
      if (!v) return;
      const el = document.getElementById('ac-' + k);
      if (el) {
        el.classList.add('sel');
        el.setAttribute('aria-checked', 'true');
        updateChk(k, true);
      }
    });
    // Subs
    Object.entries(state.subs).forEach(([k, v]) => {
      document.querySelectorAll(`.sub-btn[data-k="${k}"]`).forEach(b => {
        b.classList.toggle('sel', b.dataset.v === v);
      });
    });
    // Extras
    Object.entries(state.extras).forEach(([k, v]) => {
      if (!v) return;
      const el = document.getElementById('up-' + k);
      if (el) { el.classList.add('sel'); el.setAttribute('aria-checked','true'); }
    });
  }

  // ── PRICING ──────────────────────────────────────────────────
  const TIER = [0, 1750, 3200, 4450, 5500];

  function attrCount() {
    return Object.values(state.attractions).filter(Boolean).length;
  }

  function calcPrice() {
    const items = [];
    let total = 0;
    const add = (label, price, type='add') => { items.push({label, price, type}); total += price; };

    const n = attrCount();
    if (n > 0) {
      // First 4 attractions: tiered pricing (4+ = 5500 fixed)
      const baseCount = Math.min(n, 4);
      const base      = TIER[baseCount]; // TIER[4]=5500
      const list      = baseCount * 1750;
      const saving    = list - base;
      const packName  = baseCount >= 4 ? 'פרימיום (4)' : baseCount === 3 ? 'מורחבת (3)' : baseCount === 2 ? 'כפולה (2)' : 'בסיס (1)';
      add(`${baseCount} אטרקציות - חבילת ${packName}`, base, 'base');
      if (saving > 0) add('חיסכון חבילה', -saving, 'discount');
      // Extra attractions beyond 4 - charged individually at full price
      if (n > 4) {
        const extras = n - 4;
        add(`${extras} אטרקציות נוספות (מעל לחבילה)`, extras * 1750);
      }
    }

    // Sparks extra
    if (state.attractions.sparks && state.subs.sparks === '8u') add('זיקוקים קרים - 8 יחידות (שדרוג)', 600);

    // Location
    if (state.locationFee > 0) add('תוספת נסיעה', state.locationFee);

    // Extras
    Object.entries(state.extras).forEach(([k, v]) => {
      if (v) add(EXTRA_HEB[k], state.extraPrices[k]);
    });

    // Gift at 4+
    if (n >= 4) items.push({ label:'🎁 קליפ Highlights - מתנה', price:0, type:'gift' });

    return { items, final: total };
  }

  const EXTRA_HEB = {
    hazer: 'מכונת ערפל רחבה',
    laser: 'מערכת לייזרים מקצועית',

  // ── RENDER PRICE ──────────────────────────────────────────────
  function renderPrice() {
    const { final } = calcPrice();
    const el = document.getElementById('pb-total');
    if (!el) return;
    el.textContent = final.toLocaleString() + ' ₪';
    el.classList.add('flash');
    setTimeout(() => el.classList.remove('flash'), 400);
    const mob = document.getElementById('mob-price');
    if (mob) mob.textContent = final.toLocaleString() + ' ₪';
    saveLS();
  }

  // ── GIFT BAR ──────────────────────────────────────────────────
  function renderGiftBar() {
    const n   = attrCount();
    const pct = Math.min((n / 4) * 100, 100);
    document.getElementById('gift-fill').style.width = pct + '%';
    document.getElementById('gift-fill').parentElement.setAttribute('aria-valuenow', n);
    const txt = document.getElementById('gift-txt');
    const sub = document.getElementById('gift-sub');
    const bdg = document.getElementById('gift-badge');
    if (n >= 4) {
      txt.innerHTML = '<span style="color:var(--green);font-weight:800">🎁 מדהים! קליפ Highlights נוסף לחבילתכם בחינם!</span>';
      sub.textContent = ''; bdg.innerHTML = '<span class="badge badge-red">GIFT UNLOCKED ✦</span>';
    } else {
      const need = 4 - n;
      txt.innerHTML = `עוד <strong>${need} אטרקציה${need !== 1 ? 'ות' : ''}</strong> - וה<strong>קליפ ב-950 ₪ יהיה שלכם במתנה! 🎁</strong>`;
      sub.textContent = `נבחרו ${n}/4 · חיסכון עד 1,500 ₪ בחבילת פרימיום`;
      bdg.innerHTML = '';
    }
  }

  // ── PROGRESS BAR ──────────────────────────────────────────────
  function renderProgress() {
    const labels = ['פרטי האירוע','אטרקציות','שדרוגים','סיכום ושריון'];
    let html = '';
    for (let i = 1; i <= 4; i++) {
      const done = i < state.currentStep, active = i === state.currentStep;
      html += `<div class="pn ${done?'done':''} ${active?'active':''}" onclick="APP.goTo(${i})" tabindex="0" onkeydown="if(event.key==='Enter')APP.goTo(${i})" title="${labels[i-1]}" role="button" aria-current="${active?'step':'false'}">
        <div class="pn-num">${done ? '<svg width="10" height="10" viewBox="0 0 24 24" fill="none"><path d="M20 6L9 17l-5-5" stroke="#fff" stroke-width="3" stroke-linecap="round" stroke-linejoin="round"/></svg>' : i}</div>
        <div class="pn-lbl">${labels[i-1]}</div>
      </div>`;
      if (i < 4) html += `<div class="pn-line ${done?'done':''}"></div>`;
    }
    document.getElementById('progress').innerHTML = html;
  }

  // ── SHOW STEP ─────────────────────────────────────────────────
  function showStep(n) {
    document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
    const panel = document.getElementById('step-' + n);
    if (panel) panel.classList.add('active');
    renderProgress();
    renderPrice();
    renderGiftBar();
    // Wedding cross-sell
    const wcs = document.getElementById('wedding-cs');
    if (wcs) wcs.style.display = (state.eventType === 'wedding' && n === 3) ? 'block' : 'none';
    // Also-chose
    renderAlsoChose(n);
    if (n === 4) renderSummary();
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  // ── NAVIGATION ────────────────────────────────────────────────
  // Inline field error helper (module-scoped so no hoisting issue)
  function showFieldErr(id) {
    const el = document.getElementById(id);
    if (el) {
      el.classList.add('err');
      el.addEventListener('input', () => {
        el.classList.remove('err');
        const m = document.getElementById(id + '-err');
        if (m) m.classList.remove('err-visible');
      }, { once: true });
    }
    const errEl = document.getElementById(id + '-err');
    if (errEl) errEl.classList.add('err-visible');
  }

  function next() {
    if (state.currentStep === 1) {
      const name  = document.getElementById('c-name-1')?.value.trim() || '';
      const phone = document.getElementById('c-phone-1')?.value.trim() || '';
      let hasErr = false;

      if (!state.eventType) {
        flashAlert('אנא בחרו סוג אירוע לפני המשך ⚠️'); return;
      }
      if (!state.eventDate) {
        const el = document.getElementById('ev-date');
        if (el) { el.classList.add('err'); setTimeout(()=>el.classList.remove('err'), 1500); }
        flashAlert('אנא בחרו תאריך האירוע 📅'); return;
      }
      if (!name) {
        showFieldErr('c-name-s1');
        hasErr = true;
      }
      if (!phone) {
        showFieldErr('c-phone-s1');
        hasErr = true;
      } else if (!validatePhone(phone)) {
        showFieldErr('c-phone-s1');
        const el = document.getElementById('c-phone-1');
        if (el) { el.classList.add('err'); setTimeout(()=>el.classList.remove('err'), 2000); }
        hasErr = true;
        flashAlert('מספר טלפון לא תקין ⚠️'); return;
      }
      if (hasErr) { flashAlert('אנא מלאו שם וטלפון לפני המשך ⚠️'); return; }
      // Sync contact state from step 1 fields
      state.contact.name  = name;
      state.contact.phone = phone;
      saveLS();
    }
    if (state.currentStep === 2 && attrCount() === 0) {
      flashAlert('אנא בחרו לפחות אטרקציה אחת 🎭'); return;
    }
    if (state.currentStep < 4) { state.currentStep++; showStep(state.currentStep); }
  }

  function prev() {
    if (state.currentStep > 1) { state.currentStep--; showStep(state.currentStep); }
  }

  function goTo(n) {
    if (n <= state.currentStep) { state.currentStep = n; showStep(n); }
  }

  function mobileNext() { next(); }

  function toggleAttr(key, el) {
    state.attractions[key] = !state.attractions[key];
    el.classList.toggle('sel');
    el.setAttribute('aria-checked', String(state.attractions[key]));
    updateChk(key, state.attractions[key]);
    pricePop(state.attractions[key] ? 1750 : -1750);
    renderPrice();
    renderGiftBar();
    saveLS();
  }

  function setSub(e, key, val) {
    e.stopPropagation();
    state.subs[key] = val;
    document.querySelectorAll(`.sub-btn[data-k="${key}"]`).forEach(b => b.classList.toggle('sel', b.dataset.v === val));
    renderPrice();
    saveLS();
  }

  // ── "לקוחות דומים בחרו גם" ───────────────────────────────────
  function renderAlsoChose(step) {
    if (step !== 3) return;
    const box = document.getElementById('also-chose');
    const txt = document.getElementById('also-chose-txt');
    if (!box || !txt) return;
    const selected = Object.entries(state.attractions).filter(([,v])=>v).map(([k])=>k);
    const suggestions = [];
    if (selected.includes('smoke') && !selected.includes('sparks'))      suggestions.push('זיקוקים קרים (מרהיב עם עשן כבד!)');
    if (selected.includes('sparks') && !selected.includes('smoke'))      suggestions.push('עשן כבד לסלאו (שילוב קלאסי!)');
    if (selected.includes('backstage') && !selected.includes('confetti'))suggestions.push('גשם קונפטי (הרגע המצולם!)');
    if (selected.includes('co2') && !selected.includes('laser'))         suggestions.push('מערכת לייזרים (מכפיל אפקט!)');
    if (selected.length >= 2 && !state.extras.hazer)                     suggestions.push('מכונת ערפל - ריקוד על עננים');
    if (suggestions.length === 0) { box.style.display='none'; return; }
    txt.textContent = suggestions.slice(0,2).join(' · ');
    box.style.display = 'block';
  }

  // ── PRICE POP ─────────────────────────────────────────────────
  function pricePop(diff) {
    if (!diff) return;
    const el = document.createElement('div');
    el.className = 'price-pop';
    el.textContent = (diff > 0 ? '+' : '') + diff.toLocaleString() + ' ₪';
    el.style.color = diff > 0 ? 'var(--red)' : 'var(--green)';
    document.body.appendChild(el);
    setTimeout(() => el.remove(), 1300);
  }

  // ── TOAST ─────────────────────────────────────────────────────
  function toast(msg, type='info', dur=3500) {
    const w = document.getElementById('toast-wrap');
    const el = document.createElement('div');
    el.className = `toast ${type}`;
    const icons = { success:'✅', error:'⚠️', info:'ℹ️' };
    el.innerHTML = `<span>${icons[type]||''}</span><span>${msg}</span>`;
    w.appendChild(el);
    setTimeout(() => { el.style.animation='tOut .3s ease forwards'; setTimeout(()=>el.remove(),300); }, dur);
  }

  // ── PHONE VALIDATION ──────────────────────────────────────────
  function validatePhone(phone) {
    const clean = phone.replace(/[\s\-\(\)]/g, '');
    return /^(0[5][0-9]{8}|0[2-4,8-9][0-9]{7}|\+972[5][0-9]{8})$/.test(clean);
  }

  // ── BUILD FULL DETAILS ────────────────────────────────────────
  function buildDetails() {
    const { items, final } = calcPrice();
    const n = attrCount();
    const attrList = Object.entries(state.attractions).filter(([,v])=>v).map(([k])=>{
      let s='• '+ATTR_HEB[k];
      const sub=state.subs[k];
      if(sub) s+=` - ${SUB_HEB[k+':'+sub]||sub}`;
      if(k==='confetti'){const p=state.subs['confetti-paper'];if(p) s+=` · נייר: ${SUB_HEB['confetti-paper:'+p]||p}`;}
      return s;
    }).join('\n');
    const extrasList = Object.entries(state.extras).filter(([,v])=>v).map(([k])=>'• '+EXTRA_HEB[k]).join('\n')||'ללא';
    const csText = Object.entries(state.crossSell).filter(([,v])=>v).map(([k])=>k==='photo'?'צלם פרימיום':'דיג\'יי מומחה').join(', ')||'ללא';
    const breakdown = items.map(it=>`${it.label}: ${it.price>0?'+':it.price<0?'−':''}${Math.abs(it.price).toLocaleString()} ₪`).join('\n');
    return { attrList, extrasList, csText, breakdown, final, n };
  }
