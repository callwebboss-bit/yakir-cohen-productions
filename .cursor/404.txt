        });
      })
      .catch(function() { /* silent */ });

    /* -- 7. SEARCH UI ----------------------------------------- */
    var input    = document.getElementById('search-input');
    var dropdown = document.getElementById('search-dropdown');
    var srAnnounce = document.getElementById('srAnnounce');

    function esc(str) {
      return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    }
    function hl(text, q) {
      if (!q) return text;
      return text.replace(new RegExp('(' + esc(q) + ')', 'gi'), '<mark>$1</mark>');
    }

    function openDropdown(html) {
      dropdown.innerHTML = html;
      dropdown.classList.add('open');
    }
    function closeDropdown() {
      dropdown.classList.remove('open');
      dropdown.innerHTML = '';
    }

    /* zero-state on focus */
    input.addEventListener('focus', function() {
      if (!input.value.trim()) showZeroState();
    });

    function showZeroState() {
      var chips = POPULAR.map(function(q) {
        return '<button class="chip" data-q="' + q + '">' + q + '</button>';
      }).join('');
      openDropdown(
        '<div class="zero-state">' +
          '<span class="zero-label">\u05D7\u05D9\u05E4\u05D5\u05E9\u05D9\u05DD \u05E0\u05E4\u05D5\u05E6\u05D9\u05DD</span>' +
          '<div class="chips">' + chips + '</div>' +
        '</div>');
    }

    /* chip click */
    dropdown.addEventListener('click', function(e) {
      var chip = e.target.closest('.chip');
      if (chip) {
        input.value = chip.dataset.q;
        runSearch(chip.dataset.q);
      }
    });

    /* debounced input */
    var timer;
    input.addEventListener('input', function() {
      clearTimeout(timer);
      var val = input.value.trim();
      if (!val) { showZeroState(); return; }
      timer = setTimeout(function() { runSearch(val); }, 200);
    });

    function runSearch(query) {
      var q = query.toLowerCase();
      var results = searchIndex
        .filter(function(item) {
          return (item.title || '').toLowerCase().indexOf(q) !== -1 ||
                 (item.description || '').toLowerCase().indexOf(q) !== -1 ||
                 (item.keywords || []).some(function(k) { return k.toLowerCase().indexOf(q) !== -1; });
        })
        .slice(0, 8);

      /* target="_top" on all search result links */
      var rows = results.map(function(item) {
        return '<a class="result-row" href="' + item.url + '" target="_top" role="option">' +
          '<div class="result-info">' +
            '<span class="result-title">' + hl(item.title, query) + '</span>' +
            '<span class="result-cat">' + item.category + '</span>' +
          '</div>' +
          '<svg class="result-arrow" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">' +
            '<polyline points="15 18 9 12 15 6"/>' +
          '</svg>' +
        '</a>';
      }).join('');

      var waText = encodeURIComponent('\u05D4\u05D9\u05D9 \u05D9\u05E7\u05D9\u05E8, \u05D7\u05D9\u05E4\u05E9\u05EA\u05D9 "' + query + '" \u05D1\u05D0\u05EA\u05E8 \u05D5\u05DC\u05D0 \u05DE\u05E6\u05D0\u05EA\u05D9. \u05D0\u05E4\u05E9\u05E8 \u05E2\u05D6\u05E8\u05D4?');
      var waRow =
        '<div class="result-wa">' +
          '<p>' + (results.length === 0 ? '\u05DC\u05D0 \u05DE\u05E6\u05D0\u05E0\u05D5 \u05EA\u05D5\u05E6\u05D0\u05D5\u05EA \u05E2\u05D1\u05D5\u05E8 "' + query + '"' : '\u05DC\u05D0 \u05DE\u05E6\u05D0\u05EA \u05DE\u05D4 \u05E9\u05D7\u05D9\u05E4\u05E9\u05EA?') + '</p>' +
          '<a href="' + WA_BASE + waText + '" target="_blank" rel="noopener">' +
            '<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"/></svg>' +
            '\u05E9\u05D0\u05DC \u05D0\u05EA \u05D9\u05E7\u05D9\u05E8 \u05D9\u05E9\u05D9\u05E8\u05D5\u05EA \u05D1-WhatsApp' +
          '</a>' +
        '</div>';

      openDropdown(rows + waRow);

      // Accessibility announcement
      if (srAnnounce) {
        srAnnounce.textContent = results.length > 0
          ? '\u05E0\u05DE\u05E6\u05D0\u05D5 ' + results.length + ' \u05EA\u05D5\u05E6\u05D0\u05D5\u05EA'
          : '\u05DC\u05D0 \u05E0\u05DE\u05E6\u05D0\u05D5 \u05EA\u05D5\u05E6\u05D0\u05D5\u05EA';
      }
    }

    /* close on outside click */
    document.addEventListener('click', function(e) {
      if (!e.target.closest('.search-section')) closeDropdown();
    });
    document.addEventListener('keydown', function(e) {
      if (e.key === 'Escape') closeDropdown();
    });

    /* -- 8. VOICE SEARCH -------------------------------------- */
    var voiceBtn = document.getElementById('voice-btn');
    var SR = window.SpeechRecognition || window.webkitSpeechRecognition;

    if (SR) {
      var recognition = new SR();
      recognition.lang = 'he-IL';
      recognition.interimResults = false;
      recognition.maxAlternatives = 1;

      voiceBtn.addEventListener('click', function() {
        voiceBtn.classList.add('listening');
        if (srAnnounce) srAnnounce.textContent = '\u05DE\u05D0\u05D6\u05D9\u05DF...';
        recognition.start();
      });
      recognition.onresult = function(e) {
        var transcript = e.results[0][0].transcript;
        input.value = transcript;
        runSearch(transcript);
        voiceBtn.classList.remove('listening');
      };
      recognition.onerror  = function() { voiceBtn.classList.remove('listening'); };
      recognition.onend    = function() { voiceBtn.classList.remove('listening'); };
    } else {
      voiceBtn.style.display = 'none';
    }

  })();
  </script>
