<script>
const STATE = {
  svc: null,       // 'cover' | 'ai'
  pkg: null,       // package id
  pkgName: null,
  pkgPrice: 0,
  clip: 'none',
  clipName: 'ללא קליפ',
  clipPrice: 0
};

function selectService(type) {
  STATE.svc = type;
  STATE.pkg = null; STATE.pkgName = null; STATE.pkgPrice = 0;
  STATE.clip = 'none'; STATE.clipName = 'ללא קליפ'; STATE.clipPrice = 0;

  document.getElementById('svc-cover').classList.toggle('selected', type === 'cover');
  document.getElementById('svc-ai').classList.toggle('selected', type === 'ai');

  document.getElementById('pkg-section').classList.add('show');
  document.getElementById('cover-pkgs').style.display = type === 'cover' ? 'flex' : 'none';
  document.getElementById('ai-pkgs').style.display   = type === 'ai'    ? 'flex' : 'none';
  document.getElementById('grant-box-wrap').style.display = type === 'ai' ? 'block' : 'none';

  document.getElementById('cover-name-wrap').style.display  = type === 'cover' ? 'block' : 'none';
  document.getElementById('ai-concept-wrap').style.display  = type === 'ai'    ? 'block' : 'none';

  document.getElementById('clip-section').classList.remove('show');
  document.getElementById('info-section').classList.remove('show');
  document.getElementById('sticky-bar').classList.remove('up');

  clearPkgSelection();
  clearClipSelection();
  updateSummary();

  setTimeout(() => document.getElementById('pkg-section').scrollIntoView({ behavior: 'smooth', block: 'start' }), 150);
}

function clearPkgSelection() {
  document.querySelectorAll('.pkg-card').forEach(c => c.classList.remove('selected'));
}
function clearClipSelection() {
  document.querySelectorAll('.clip-card').forEach(c => c.classList.remove('selected'));
  document.getElementById('clip-none') && document.getElementById('clip-none').classList.add('selected');
}

function selectPkg(svc, id, price, name) {
  if (STATE.svc !== svc) return;
  STATE.pkg = id; STATE.pkgName = name; STATE.pkgPrice = price;

  clearPkgSelection();
  document.getElementById(svc + '-' + id).classList.add('selected');

  document.getElementById('clip-section').classList.add('show');
  document.getElementById('info-section').classList.add('show');
  clearClipSelection();
  document.getElementById('sticky-bar').classList.add('up');
  updateSummary();

  setTimeout(() => document.getElementById('clip-section').scrollIntoView({ behavior: 'smooth', block: 'start' }), 200);
}

function selectClip(id, price, name) {
  STATE.clip = id; STATE.clipPrice = price; STATE.clipName = name;
  document.querySelectorAll('.clip-card').forEach(c => c.classList.remove('selected'));
  document.getElementById('clip-' + id).classList.add('selected');
  updateSummary();
}

function updateSummary() {
  const total = STATE.pkgPrice + STATE.clipPrice;
  document.getElementById('sum-svc').textContent  = STATE.svc === 'cover' ? 'הקלטת קאבר' : STATE.svc === 'ai' ? 'שיר מקורי + AI' : '-';
  document.getElementById('sum-pkg').textContent  = STATE.pkgName || '-';
  document.getElementById('sum-clip').textContent = STATE.clipName || '-';
  document.getElementById('sum-total').textContent = total > 0 ? '₪' + total.toLocaleString() : '₪0';
  document.getElementById('wa-btn-text').textContent = total > 0 ? 'שריינ/י עכשיו - ₪' + total.toLocaleString() : 'שליחה ליקיר בוואצאפ';
}

function sendToWA(e) {
  e.preventDefault();

  const name   = document.getElementById('fname').value.trim();
  const phone  = document.getElementById('fphone').value.trim();
  const notes  = document.getElementById('fnotes').value.trim();
  const cover  = document.getElementById('fcovername') ? document.getElementById('fcovername').value.trim() : '';
  const concept= document.getElementById('fconcept') ? document.getElementById('fconcept').value.trim() : '';

  if (!name || !phone) {
    alert('יש להכניס שם ומספר טלפון.');
    document.getElementById('info-section').scrollIntoView({ behavior: 'smooth' });
    return;
  }
  if (!STATE.pkg) {
    alert('יש לבחור חבילה.');
    return;
  }

  const total = STATE.pkgPrice + STATE.clipPrice;
  const svcLabel = STATE.svc === 'cover' ? 'הקלטת קאבר' : 'שיר מקורי + AI';

  let msg = 'היי יקיר! סיימתי את הטופס ורוצה לקדם 🎤\n\n';
  msg += '*סוג שירות:* ' + svcLabel + '\n';
  msg += '*חבילה:* ' + STATE.pkgName + ' - ₪' + STATE.pkgPrice.toLocaleString() + '\n';
  msg += '*קליפ:* ' + STATE.clipName + (STATE.clipPrice > 0 ? ' (+₪' + STATE.clipPrice.toLocaleString() + ')' : '') + '\n';
  msg += '*סה"כ:* ₪' + total.toLocaleString() + '\n\n';
  msg += '*שם:* ' + name + '\n';
  msg += '*טלפון:* ' + phone + '\n';
  if (cover)   msg += '*שיר לקאבר:* ' + cover + '\n';
  if (concept) msg += '*קונספט השיר:* ' + concept + '\n';
  if (notes)   msg += '*הערות:* ' + notes + '\n';

  const encoded = encodeURIComponent(msg);
  window.open('https://wa.me/972587555456?text=' + encoded, '_blank');
}
