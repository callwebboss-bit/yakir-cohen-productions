const APP = (() => {

  // ── STATE ──────────────────────────────────────────────────────
  const state = {
    currentStep: 1,
    eventType:   null,
    eventLabel:  '',
    eventDate:   '',
    guests:      0,
    locationFee: 0,
    isOutdoor:   false,
    startTime:   '',
    hours:       8,
    // Core services (step 2)
    services: {
      video:   false,
      magnets: false
    },
    // Boutique upsells (step 4)
    upsells: {
      prelude:    false,
      avantgarde: false,
      reels:      false,
      priority:   false,
      archive:    false
    },
    // AI upsells (step 4, separate)
    aiUpsells: {
      aerial:  false,
      retouch: false,
      cinema:  false
    },
    contact: { name:'', phone:'', email:'', notes:'' }
  };

  // ── PRICING ────────────────────────────────────────────────────
  const PRICE = {
    hourRate:  980,
    services: {
      video:   2500,
      magnets: 1800
    },
    upsells: {
      prelude:    2200,
      avantgarde: 3800,
      reels:      2800,
      priority:   1400,
      archive:    2800
    },
    ai: {
      aerial:  850,
      retouch: 1200,
      cinema:  950
    },
    aiBundle:       -500,  // discount for 2+ AI services
    pkgThreshold:  15000,
    pkgDiscount:    0.12
  };

  const LABELS = {
    services: {
      video:   'וידאו קולנועי (4K Cinematic)',
      magnets: 'מגנטי Fine-Art (איכות גלריה)'
    },
    upsells: {
      prelude:    'הפתיח – צילומי זוגיות / Save the Date',
      avantgarde: 'אוונגרד – Trash the Dress אמנותי',
      reels:      'עורך הסיפור – Reels בזמן אמת',
      priority:   'מסירה מהירה – תוך 48 שעות',
      archive:    'ארכיון מאסטר – גיבוי 10 שנים'
    },
    ai: {
      aerial:  'פנורמות AI (תחליף רחפן)',
      retouch: 'ריטוש מאסטר AI',
      cinema:  'קליפ AI (מוכן לבוקר שאחרי)'
    }
  };

  // ── PRICING CALCULATOR ─────────────────────────────────────────
  function calcPrice() {
    const items = [];
    let subtotal = 0;
    const add = (label, price, type = 'add') => {
      items.push({ label, price, type });
      subtotal += price;
    };

    // Hours × rate
    const hoursCost = state.hours * PRICE.hourRate;
    add(`${state.hours} שעות צילום (${PRICE.hourRate} ₪/שעה)`, hoursCost, 'base');

    // Location surcharge
    if (state.locationFee > 0) add('תוספת נסיעה', state.locationFee);

    // Core services
    for (const [key, active] of Object.entries(state.services)) {
      if (active) add(LABELS.services[key], PRICE.services[key]);
    }

    // Boutique upsells
    for (const [key, active] of Object.entries(state.upsells)) {
      if (active) add(LABELS.upsells[key], PRICE.upsells[key]);
    }

    // AI upsells
    const aiCount = Object.values(state.aiUpsells).filter(Boolean).length;
    for (const [key, active] of Object.entries(state.aiUpsells)) {
      if (active) add(LABELS.ai[key], PRICE.ai[key], 'ai');
    }

    // AI Bundle discount
    if (aiCount >= 2) {
      add('AI Bundle Reward', PRICE.aiBundle, 'discount');
    }

    // Package discount (12% over 15,000)
    let pkgDiscount = 0;
    if (subtotal > PRICE.pkgThreshold) {
      pkgDiscount = Math.round(subtotal * PRICE.pkgDiscount);
      add('הנחת חבילה 12%', -pkgDiscount, 'discount');
      subtotal -= pkgDiscount;
    }

    return { items, final: subtotal, pkgDiscount };
  }

  // ── RENDER PRICE BAR ───────────────────────────────────────────
  function renderPrice() {
    const { final, pkgDiscount } = calcPrice();
    const el = document.getElementById('pb-total');
    el.textContent = final.toLocaleString() + ' ₪';
    el.classList.add('flash');
    setTimeout(() => el.classList.remove('flash'), 400);

  // ── DISCOUNT GAMIFICATION BAR ──────────────────────────────────
  function renderDiscBar() {
    const { final, pkgDiscount } = calcPrice();
    const wrap = document.getElementById('disc-wrap');
    if (final < 1000) { wrap.style.display = 'none'; return; }
    wrap.style.display = 'block';
    const pct = Math.min((final / PRICE.pkgThreshold) * 100, 100);
    document.getElementById('disc-fill').style.width = pct + '%';
    const txt  = document.getElementById('disc-txt');
    const sub  = document.getElementById('disc-sub');
    const bdg  = document.getElementById('disc-badge');
    if (final >= PRICE.pkgThreshold) {
      txt.innerHTML = `<span style="color:var(--green);font-weight:800">✓ הנחת חבילה 12% פועלת - חסכתם ${pkgDiscount.toLocaleString()} ₪!</span>`;
      sub.textContent = '';
      bdg.textContent = '';
    } else {
      const need = PRICE.pkgThreshold - final;
      txt.innerHTML = `עוד <strong>${need.toLocaleString()} ₪</strong> לפתיחת הנחת חבילה 12%`;
      sub.textContent = `שווי הנחה פוטנציאלי: כ-${Math.round(PRICE.pkgThreshold * PRICE.pkgDiscount).toLocaleString()} ₪`;
      bdg.textContent = '';
    }
  }

  // ── STEP VISIBILITY ────────────────────────────────────────────
  function showStep(n) {
    document.querySelectorAll('.step-panel').forEach(p => p.classList.remove('active'));
    const panel = document.getElementById('step-' + n);
    if (panel) panel.classList.add('active');

  // ── NAVIGATION ─────────────────────────────────────────────────
  function next() {
    const s = state.currentStep;
    // Validate
    if (s === 1 && !state.eventType) {
      flashAlert('אנא בחרו סוג אירוע לפני המשך ⚠️');
      return;
    }
    if (s === 1 && state.isOutdoor && !state.startTime) {
      flashAlert('לאירוע חוץ נא לציין שעת התחלה ⏰');
      return;
    }
    if (s < 5) {
      state.currentStep++;
      showStep(state.currentStep);
    }
  }

  function prev() {
    if (state.currentStep > 1) {
      state.currentStep--;
      showStep(state.currentStep);
    }
  }

  function goTo(n) {
    if (n <= state.currentStep) {
      state.currentStep = n;
      showStep(n);
    }
  }

  function setPreset(hours, id) {
    state.hours = hours;
    // Sync slider
    const slider = document.getElementById('hours-slider');
    if (slider) slider.value = hours;
    updateSliderReadout(hours);
    // Highlight preset card
    ['essence','chronicle','legacy'].forEach(name => {

  function onSlider(val) {
    const hours = parseInt(val);
    state.hours = hours;
    updateSliderReadout(hours);
    // Deselect presets if custom value
    const presetMap = { 4:'essence', 8:'chronicle', 12:'legacy' };
    ['essence','chronicle','legacy'].forEach(name => {
      const el = document.getElementById('preset-' + name);
      if (el) {
        const match = presetMap[hours] === name;
        el.classList.toggle('sel', match);
        el.setAttribute('aria-checked', String(match));
      }
    });
    renderPrice();
    renderDiscBar();
  }

      // Table

      doc.setFillColor(34, 34, 34);
      doc.rect(10, y - 5, 190, 10, 'F');
      doc.setTextColor(255, 255, 255);
      doc.setFont('helvetica', 'bold'); doc.setFontSize(9);
      doc.text('Service / Item', 175, y + 1, { align: 'right' });
      doc.text('Price (ILS)', 35, y + 1, { align: 'left' });
      y += 10;

      const { items, final, pkgDiscount } = calcPrice();
      items.forEach((it, i) => {
        if (y > 265) { doc.addPage(); y = 20; }
        if (i % 2 === 0) { doc.setFillColor(248, 247, 245); doc.rect(10, y - 4, 190, 9, 'F'); }
        const neg = it.price < 0;
        doc.setTextColor(neg ? 26 : 51, neg ? 107 : 51, neg ? 60 : 51);
        doc.setFont('helvetica', 'normal'); doc.setFontSize(8.5);
        doc.text(it.label, 175, y + 1, { align: 'right' });
        const ps = (neg ? '-' : '+') + Math.abs(it.price).toLocaleString() + ' ILS';
        doc.text(ps, 35, y + 1, { align: 'left' });
        y += 9;
      });

      // Total bar
      y += 4;

    renderProgress();
    // Default: 8 hours preset selected - renders 7,840 ₪ immediately
    setPreset(8, 'chronicle');
    renderPrice();
    // Close modal on overlay click
    document.getElementById('modal-overlay').addEventListener('click', e => {
      if (e.target === document.getElementById('modal-overlay')) closeModal();
    });
    // Update slider gradient live
    const slider = document.getElementById('hours-slider');
    if (slider) {
      const update = () => {
        const pct = ((slider.value - 1) / (slider.max - 1)) * 100;
        slider.style.background = `linear-gradient(to right, var(--burg) ${pct}%, var(--border) ${pct}%)`;
      };
      slider.addEventListener('input', update);
      update();
    }
  }

  // ── PUBLIC API ────────────────────────────────────────────────
  return {
    state,
    next, prev, goTo,
    pickEvent,
    toggleSvc,
    setPreset, onSlider,
    toggleUp, toggleAI,
    runAlerts, closeAlert,
    checkAllLegal,
    submitForm, sendWhatsApp, downloadPDF, copyQuote,
    closeModal,
    mobileNext,
    render: renderPrice,
    init
  };

})();

document.addEventListener('DOMContentLoaded', () => APP.init());
