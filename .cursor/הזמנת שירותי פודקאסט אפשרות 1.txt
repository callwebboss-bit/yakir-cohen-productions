/* ════ ANIMATE VALUE ════ */
function animVal(el,from,to,dur){
  if(raf) cancelAnimationFrame(raf);
  if(from===to){el.textContent='₪'+to.toLocaleString();return;}
  let ts=null;
  const step=t=>{
    if(!ts) ts=t;
    const p=Math.min((t-ts)/dur,1),e=1-Math.pow(1-p,3);
    el.textContent='₪'+Math.floor(e*(to-from)+from).toLocaleString();
    if(p<1){raf=requestAnimationFrame(step);}else{el.textContent='₪'+to.toLocaleString();raf=null;}
  };
  raf=requestAnimationFrame(step);
}

/* ════ ISRAELI PHONE VALIDATION ════ */
function validIsraeliPhone(p) {
  const clean = (p||'').replace(/[\s\-\(\)]/g,'');
  return /^(0[5][0-9]{8}|0[2-4,8-9][0-9]{7}|\+972[5][0-9]{8})$/.test(clean);
}
function validateContactFields() {
  const nameEl = document.getElementById('cfName');
  const contEl = document.getElementById('cfContact');
  const name = (nameEl?.value||'').trim();
  const cont = (contEl?.value||'').trim();
  let valid = true;
  if (nameEl && name) {
    const ok = name.length >= 2;
    nameEl.style.borderColor = ok ? '' : 'var(--red)';
    if (!ok) valid = false;
  }
  if (contEl && cont) {
    const isPhone = validIsraeliPhone(cont);
    const isEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(cont);
    const ok = isPhone || isEmail;
    contEl.style.borderColor = ok ? '' : 'var(--red)';
    if (!ok) { contEl.title = 'מספר טלפון ישראלי או אימייל תקין'; valid = false; }
  }
  return valid;
}
